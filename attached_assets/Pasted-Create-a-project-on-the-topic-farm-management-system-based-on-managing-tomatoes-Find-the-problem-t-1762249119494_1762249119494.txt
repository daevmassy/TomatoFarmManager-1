â˜• Sample Java Servlet (AddPlantServlet.java)
import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
import java.sql.*;

public class AddPlantServlet extends HttpServlet {
  protected void doPost(HttpServletRequest request, HttpServletResponse response)
      throws ServletException, IOException {
    String variety = request.getParameter("variety");
    String plantingDate = request.getParameter("planting_date");
    String harvestDate = request.getParameter("harvest_date");

    try {
      Class.forName("com.mysql.jdbc.Driver");
      Connection con = DriverManager.getConnection("jdbc:mysql://localhost/farmdb", "root", "password");
      PreparedStatement ps = con.prepareStatement("INSERT INTO tomato_plants(variety, planting_date, expected_harvest_date) VALUES (?, ?, ?)");
      ps.setString(1, variety);
      ps.setString(2, plantingDate);
      ps.setString(3, harvestDate);
      ps.executeUpdate();
      response.sendRedirect("success.html");
    } catch (Exception e) {
      e.printStackTrace();
    }
  }
}
